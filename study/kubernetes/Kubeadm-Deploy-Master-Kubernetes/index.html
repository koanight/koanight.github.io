通过Kubeadm部署Kubernetes单主多从集群

本文最後更新於：20 天前

学习环境

vmware安装三台虚拟机，或者先安装一台，完成下面的准备操作后克隆两台虚拟机，再修改IP；主机名可以在安装时候设置，或者装完之后通过hostnamectl set-hostname修改；

hostnameosstatic iprole
mastercentos7.6192.168.204.131master
minioncentos7.6192.168.204.132node1
slavecentos7.6192.168.204.133node2

没有特殊说明，默认每台虚拟机都要操作；

网站文档：https://kubernetes.io/zh/docs/setup/production-environment/tools/kubeadm/install-kubeadm/；

准备操作

#临时关闭
systemctl stop firewalld
#禁用，重启生效
systemctl disable firewalld

#临时关闭
setenforce 0
#禁用,重启生效
sed -i "s/enforcing/disabled/g" /etc/selinux/config

#临时关闭
swapoff -a
#永久关闭
sed -i 's/^[^#].*swap*/#&/g' /etc/fstab

#安装常用软件包
yum install vim net-tools lrzsz unzip dos2unix telnet sysstat iotop pciutils lsof tcpdump psmisc bc wget socat gcc tree chrony ntpdate mlocate -y

#主机名写入hosts
cat >> /etc/hosts << EOF
192.168.204.131 master
192.168.204.132 minion
192.168.204.133 slave
EOF

#主机间配置ssh免密登录
ssh-keygen -t rsa
for host in master minion slave; do ssh-copy-id -i ~/.ssh/id_rsa.pub $host;done

#内核开启网络支持
cat > /etc/sysctl.d/kubernetes.conf << EOF
net.bridge.bridge-nf-call-ip6tables = 1
net.bridge.bridge-nf-call-iptables = 1
net.ipv4.ip_forward = 1
net.ipv4.ip_nonlocal_bind = 1
EOF
modprobe br_netfilter
sysctl -p /etc/sysctl.d/kubernetes.conf

Docker

#官方脚本安装
curl -fsSL https://get.docker.com | bash -s docker --mirror Aliyun

#手动yum安装
#卸载原有docker
yum remove docker*
#安装所需的软件包。yum-utils 提供了 yum-config-manager ，并且 device mapper 存储驱动程序需要 device-mapper-persistent-data 和 lvm2
yum install -y yum-utils device-mapper-persistent-data lvm2
#设置阿里源地址
yum-config-manager --add-repo http://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo
#安装
yum install docker-ce docker-ce-cli containerd.io -y

#配置镜像加速地址,并修改Cgroup Driver(docker默认cgroupfs，k8s推荐使用systemd)
mkdir /etc/docker -p
cat > /etc/docker/daemon.json << EOF
{"registry-mirrors":["https://docker.mirrors.ustc.edu.cn/"],
  "exec-opts": ["native.cgroupdriver=systemd"]}
EOF
systemctl daemon-reload
#启动docker并设置自启
for op in enable start status;do systemctl $op docker;done

Master

#配置k8s阿里源地址
cat <<EOF > /etc/yum.repos.d/kubernetes.repo
[kubernetes]
name=Kubernetes
baseurl=http://mirrors.aliyun.com/kubernetes/yum/repos/kubernetes-el7-x86_64
enabled=1
gpgcheck=0
repo_gpgcheck=0
gpgkey=http://mirrors.aliyun.com/kubernetes/yum/doc/yum-key.gpg
http://mirrors.aliyun.com/kubernetes/yum/doc/rpm-package-key.gpg
EOF

#安装k8s相关
yum install -y kubelet kubeadm kubectl --disableexcludes=kubernetes
systemctl enable --now kubelet

#初始化master
A:参数初始(可选参数较多，命令会比较长)
kubeadm init \
--apiserver-advertise-address=192.168.204.131 \
--kubernetes-version v1.20.4 \
--service-cidr=192.100.0.0/16 \
--pod-network-cidr=192.244.0.0/16 \
--control-plane-endpoint 192.168.204.200:8443 \
--image-repository registry.cn-hangzhou.aliyuncs.com/google_containers

B:配置文件初始(通过配置文件修改参数，复用性高，易修改)
#导出默认配置文件
kubeadm config print init-defaults > kubeadm-init.yml
vim kubeadm-init.yml
//
apiVersion: kubeadm.k8s.io/v1beta2
bootstrapTokens:
- groups:
  - system:bootstrappers:kubeadm:default-node-token
  token: abcdef.0123456789abcdef                        #token,默认值即可
  ttl: 2400h0m0s                                         #token有效期，添加节点如果token过期需要重新⽣成
  usages:
  - signing
  - authentication
kind: InitConfiguration
localAPIEndpoint:
  advertiseAddress: 192.168.204.131                   #修改为本地主机IP
  bindPort: 6443
nodeRegistration:
  criSocket: /var/run/dockershim.sock
  name: master                                        #默认为本地主机名
  taints:
  - effect: NoSchedule
    key: node-role.kubernetes.io/master
---
apiServer:
  timeoutForControlPlane: 4m0s
apiVersion: kubeadm.k8s.io/v1beta2
certificatesDir: /etc/kubernetes/pki
clusterName: kubernetes
controlPlaneEndpoint: "192.168.204.131:6443"                 #单master节点时这⾥写“本地真实IP:6443”；多master节点时写“VIP:端⼝”，端⼝要与haproxy配置中的bind字段的端⼝⼀致
controllerManager: {}
dns: {}
etcd:
  local:
    dataDir: /var/lib/etcd
imageRepository: registry.cn-hangzhou.aliyuncs.com/google_containers    #镜像地址，默认为k8s地址，此处已修改为阿里云
kind: ClusterConfiguration
kubernetesVersion: v1.20.4                                    #指定k8s版本，可以通用kubectl version查看获取
networking:
  dnsDomain: cluster.local
  podSubnet: 192.244.0.0/16                                    #指定Pod的⽹络范围                        
  serviceSubnet: 192.100.0.0/16                                #指定Service的⽹络范围
scheduler: {}
//

#预下载镜像
kubeadm config images pull --config kubeadm-init.yml

#初始化
kubeadm init --config kubeadm-init.yml

#配置kubectl的config⽂件(此处命令在master初始化会有)
mkdir -p $HOME/.kube
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config

#安装flannel网络插件（k8s支持很多网络插件，此处以flannel为例）
cat > /run/flannel/subnet.env << EOF
FLANNEL_NETWORK=10.244.0.0/16
FLANNEL_SUBNET=10.244.0.1/24
FLANNEL_MTU=1450
FLANNEL_IPMASQ=true
EOF

kubectl apply -f https://raw.githubusercontent.com/coreos/flannel/2140ac876ef134e0ed5af15c65e414cf26827915/Documentation/kube-flannel.yml

#配置命令补全
yum install -y bash-completion
source /usr/share/bash-completion/bash_completion
source <(kubectl completion bash)

#切换默认命名空间
kubectl config set-context $(kubectl config current-context) --namespace=kube-system

#查看pod情况
kubectl get pod -o wide

#初始化失败或者反复可以重置
kubeadm reset

Nodes

#配置k8s阿里源地址
cat <<EOF > /etc/yum.repos.d/kubernetes.repo
[kubernetes]
name=Kubernetes
baseurl=http://mirrors.aliyun.com/kubernetes/yum/repos/kubernetes-el7-x86_64
enabled=1
gpgcheck=0
repo_gpgcheck=0
gpgkey=http://mirrors.aliyun.com/kubernetes/yum/doc/yum-key.gpg
http://mirrors.aliyun.com/kubernetes/yum/doc/rpm-package-key.gpg
EOF

#安装k8s相关
yum install -y kubelet kubeadm --disableexcludes=kubernetes

#启动kubelet并设置自启
for op in enable start status;do systemctl $op kubelet;done

#加入nodes(此处命令在master初始化会有)
kubeadm join 192.168.204.131:6443 --token abcdef.0123456789abcdef
    --discovery-token-ca-cert-hash sha256:724d623dc0cde1d350ebea86b6a583c3221a2bdef735db804b145eac751037b0
#如果忘记通过以下命令查看
kubeadm token create --print-join-command

#node节点join之后，可以在master查看节点情况（如果不安装网络插件，查看的STATUS会为NOT READY）
kubectl get nodes
NAME     STATUS   ROLES                  AGE     VERSION
master   Ready    control-plane,master   32m     v1.20.4
minion   Ready    <none>                   9m      v1.20.4
slave    Ready    <none>                4m20s   v1.20.4

安装dashboard（Master）

#下载默认yaml文件
wget https://raw.githubusercontent.com/kubernetes/dashboard/v2.0.0-beta4/aio/deploy/recommended.yaml

#修改yaml
vim recommended.yaml修改NodePort 
//
kind: Service
apiVersion: v1
metadata:
  labels:
    k8s-app: kubernetes-dashboard
  name: kubernetes-dashboard
  namespace: kubernetes-dashboard
spec:
  type: NodePort                              #默认没有，添加
  ports:
    - port: 443
      targetPort: 8443
      nodePort: 30001                        #默认没有，添加，30001为自定义端口
  selector:
    k8s-app: kubernetes-dashboard
//
#启动dashboard，执行命令之后可以稍微等一会
kubectl apply -f recommended.yaml

#查看dashboard的pod
kubectl get pod -n kubernetes-dashboard | grep dashboard

#生成证书
openssl genrsa -out dashboard.key 2048 
openssl req -new -out dashboard.csr -key dashboard.key -subj '/CN=192.168.204.131'
openssl x509 -req -days 3650 -in dashboard.csr -signkey dashboard.key -out dashboard.crt
#删除原有证书
kubectl delete secret kubernetes-dashboard-certs -n kubernetes-dashboard
#通过新生成的证书创建secret
kubectl create secret generic kubernetes-dashboard-certs --from-file=dashboard.key --from-file=dashboard.crt -n kubernetes-dashboard
#删除原有pod即可（会自动创建新的pod）
kubectl delete pod <pod name> -n kubernetes-dashboard


#给所有namespace授权
kubectl create serviceaccount dashboard-serviceaccount -n kubernetes-dashboard
#创建clusterrolebinding
kubectl create clusterrolebinding dashboard-cluster-admin --clusterrole=cluster-admin --serviceaccount=kubernetes-dashboard:dashboard-serviceaccount
#获取token
kubectl get secret -n kubernetes-dashboard | grep dashboard-serviceaccount-token
kubectl describe secret <secret name>

访问https://IP:之前自定义的端口,输入上一步获取的token即可登录看到相关信息
